Global Quake API

实时全球地震监测数据流。通过 WebSocket 长连接获取低延迟的地震预警信息。

WSS wss://gqm.aloys233.top/ws
Message Format: Earthquake Update

当监测到地震或参数修正时推送。包含了震源位置、深度、震级以及台站置信度等详细工程数据。

JSON Response Example 
{
  "type": "earthquake",
  "action": "update",
  "timestamp": "2025-12-21T14:02:01.544Z",
  "data": {
    "id": "b9e988e1-2941-4427-8d32-b0b169fec815",
    "region": "Hindu Kush region, Afghanistan", // 震源区域描述
    "latitude": 36.52647,
    "longitude": 70.82434,
    "depth": 6.8018,          // 深度 (km)
    "magnitude": 3.3019,      // 震级
    "intensity": "IV",        // 预估烈度
    "maxPGA": 5.2668,         // 最大地面峰值加速度
    "fixedDepth": false,      // 是否为人工固定深度
    
    // 时间相关 (Unix Ms & ISO)
    "originTimeMs": 1766325604853,
    "originTimeIso": "2025-12-21T14:00:04.853Z",
    "lastUpdateMs": 1766325721172,
    "revisionId": 2,          // 版本号，随数据更新递增

    // 事件簇信息 (用于关联余震)
    "cluster": {
      "id": "f614e6a3-75b5-4003...",
      "latitude": 38.8928,
      "longitude": 69.6833,
      "level": 0
    },

    // 质量分析
    "quality": {
      "pct": 100.0,           // 质量置信度 %
      "stations": 6,          // 参与解算的台站数
      "errOrigin": 3.755,     // 震中误差 (km)
      "errDepth": 4.3         // 深度误差 (km)
    },

    // 台站统计详情
    "stationCount": {
      "total": 6,             // 扫描到的总台站
      "used": 6               // 实际使用的台站
    },

    // 深度置信区间
    "depthConfidence": {
      "minDepth": 5.6018,
      "maxDepth": 9.9018
    }
  }
}
